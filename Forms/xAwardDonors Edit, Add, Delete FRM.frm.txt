Private Sub FillValueSum()
  On Error GoTo SubError
  Dim SumValue As Currency
  SumValue = 0
  With AwardsRS
    .MoveFirst
    .FindFirst "[przAN] = " & Me.AWD_NUM
    Do While Not .NoMatch
      SumValue = SumValue + IIf(IsNull(!Value), 0, !Value)
      .FindNext "[przAN] = " & Me.AWD_NUM
    Loop
  End With
  Me.ValueSum = SumValue
  Exit Sub
SubError:
  MsgBox Err.Description
End Sub

   'IMPORTANT: This and AfterUpdate not called if notinlist
   '  sets Response = acDataErrContinue (i.e. not acDataErrAdd)
   'Edit This (Delete if no cbPersons
   ' User selected a different Person or new person (with no ; and with
   ' no fixes in name string for a New record (see important note above)
   ' If an existing person, The person selected could already be
   '    referenced in another record.
  '***** NOTE:  I allow duplicates in AwardDonors Table !!!!!   ***************

  ' ALL CODE DISABLED
   On Error GoTo SubError
   ' Find the record that matches the new PID
   Dim rs As Object
   Set rs = Me.Recordset.Clone
   rs.FindFirst "[AwardDonors].[PID] = " & cbPerson.Value 'Edit this PID of selected person
   If rs.NoMatch Then

   Else
      ' Edit this (next line)

            "  (In another AwardDonors record)!" & vbCrLf & _

Private Sub cbPerson_NotInList(newdata As String, Response As Integer)
   ' If the person is not in the Persons Table then we need to add a record
   ' To the persons table.
   On Error GoTo SubError
   ' Edit This  (next Line   name of forms table and TP )
   If Person_NotInList(Me, "AwardDonors", "AD", newdata, Response, _
            DidPersonAdd, True) = True Then ' Allow Duplicates (last parameter=True)
     If Response = acDataErrAdded Then ' Person added just as typed

   End If

Private Sub btnSave_Click()
    ' Save the Indexed value so can move to it after save
    Dim SvAWD_NUM As Long   ' Replace this else Edit This
    Dim SvPID As Long       ' In this case need person also since can
                       ' be more than one contact per award
    On Error GoTo Err_btnSave_Click
    ' Check that at least the minimum is done. NOTE: any code put here will not
    ' be executed if the user does menu save Record, Shift Enter or moves to
    ' Another record - so... This is just to 'Pre-Warn'
    If IsNull(Me.AWD_NUM) Then ' Replace This else Edit This
      MsgBox "Must Enter AWD_NUM! (or Undo)", vbOKOnly, Title 'Rpl/ Edit This
      On Error Resume Next
      ctlPrevFocus.SetFocus
      Exit Sub
    Else
      SvAWD_NUM = Me.AWD_NUM  ' Replace both Else Edit This
      SvPID = 0
      SvPID = Me.cbPerson.Value
      
    End If
    If InAdd Then  ' Only save if in add mode
      DoCmd.DoMenuItem acFormBar, acRecordsMenu, acSaveRecord, , acMenuVer70
    End If
    ' The next command does a Remove Filter Sort which in effect requeries
    DoCmd.DoMenuItem acFormBar, acRecordsMenu, 3, , acMenuVer70 'RemFiltSort
    On Error Resume Next
    ctlPrevFocus.SetFocus
    If SvPID > 0 Then
     Me.Recordset.FindFirst "AWD_NUM = " & SvAWD_NUM & _
        " AND [Awarddonors].[PID] = " & SvPID
    Else
     Me.Recordset.FindFirst "AWD_NUM = " & SvAWD_NUM  'Replace Both /Edit This
    End If
Exit_btnSave_Click:
    Exit Sub
Err_btnSave_Click:
    MsgBox Err.Description
    Resume Exit_btnSave_Click
End Sub

Private Sub Type_Exit(Cancel As Integer)
    On Error GoTo SubError
    If IsNull(Me.[Type]) Then
      MsgBox "Donor Type Set to 'P'rimary " & vbCrLf & _
          "Change if not Primary Donor for this award!", vbOKOnly
      Me.Type = "P"
    End If
    If StrComp(Me.[Type], UCase(Me.[Type]), 0) <> 0 Then ' Binary compare
        Me.[Type] = UCase(Me.[Type])   ' do only if not already ucase
    End If
    Exit Sub
SubError:
    MsgBox Err.Description
End Sub
