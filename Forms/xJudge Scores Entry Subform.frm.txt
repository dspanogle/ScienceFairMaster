Private Function OKChangeMark() As Boolean
  '  Changes 5.4.1 added MsgDone and code to show message only once per form load
  If MsgDone = False Then   ' Show this only once
     If MsgBox("REALLY CHANGE? Best to re-read Score Form Instead!" & vbCrLf & _
      "Click [NO] and press [Esc] to clear your change!" & vbCrLf & vbCrLf & _
      "You should change score card marks ONLY IF YOU ARE AUTHORIZED!" & vbCrLf & _
      "Then (and only then) you may click [Yes]" & vbCrLf & vbCrLf & _
      "ARE YOU AUTHORIZED TO MAKE THIS CHANGE?", vbYesNo + vbDefaultButton2, _
      "WARNING - NOT NORMALLY CHANGED") = vbYes Then
          MsgDone = True
          OKChangeMark = True
          MsgBox "Warning disabled! Be careful with addtional changes."
      Else
        OKChangeMark = False
      End If
   Else
     OKChangeMark = True
   End If
   Exit Function
End Function
