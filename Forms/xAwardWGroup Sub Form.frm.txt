Private Sub cbGrpNum_Enter()
' Rev 5.4.0   add group numbers to dropdown box
On Error GoTo ErrorExit
Dim rs As Recordset
' rev 5.4.3 543  changed next three from integer to Long as GID can be as big as 99999
Dim iGN As Long
Dim an100 As Long
Dim iGID As Long
Set rs = CurrentDb.OpenRecordset("Select GID from AwardGroup where AN = " & Me.AN, dbOpenDynaset)
Me.cbGrpNum.RowSource = ""  ' clear out old values
Do While Not rs.EOF
  iGID = rs!GID
  an100 = iGID / 100
  an100 = 100 * an100
  iGN = iGID - an100
  Me.cbGrpNum.AddItem (iGN)   ' fill in new values
  rs.MoveNext
Loop
Exit Sub
ErrorExit:
MsgBox "Grpnum box list " & Err.Description
End Sub

Private Sub Form_Delete(Cancel As Integer)
   ' This plus Sub Form_BeforeDelConfirm work to ALWAYS show a delete message
   ' Even if user has turned off the warning messages.
   On Error GoTo SubError
   msg = "CAUTION: Deleting a Group will delete all" & vbCrLf & _
         "prizes for that Group " & vbCrLf & vbCrLf & _
         "DELETE GROUP / Prizes?"
   If MsgBox(msg, vbYesNo + vbDefaultButton2, Title) = vbNo Then
      Cancel = True
      Exit Sub
   End If
   DidDelete = True   ' Indicate did a delete of the judge
   GIDval = Me.GID       ' save GID (AN and GN)  num for log
   TANam = IIf(IsNull([G_Desc]), " ", Mid([G_Desc], 1, 49))  '  for log
   Exit Sub  ' Judge record will be deleted on exit
SubError:
   MsgBox Err.Description
End Sub
