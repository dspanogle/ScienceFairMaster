SELECT ExhibitStudents.F_ORD AS [Select&SortOnlyFirstStudent], Persons.LNAME & ", " & Persons.FNAME AS Student, ExhibitData.EXN AS Exhibit, IIf(Left([NCAT],1)="S","Senior","Junior") AS DIV, ExhibitData.NCAT AS NCAT, ExhibitData.JCAT, Mid([NCAT],2,5) AS Category, ExhibitData.EXHB_TITLE, ExhibitData.SCH_CODE, Schools.SCHOOL, IIf([FLOOR]="Y","FLOOR!!",IIf([FLOOR]="N","TABLE","?????")) AS FloorTbl, ExhibitData.PAID, ExhibitData.PROTOID, Persons_1.LNAME AS TLN, Persons_1.SALUTATION & " " & [Persons_1.FNAME] & " " & [Persons_1.MI] & " " & [Persons_1.LNAME] AS TeacherName, Schools.PHONE AS SCH_Phone, Schools.ADDR & ",  " & Schools.CITY & "  " & Schools.STATE & ",  " & Schools.ZIP AS SchAdr, Students.LAST_FAIR, Categories.TEAM, Categories.CATNAME, Chairs.GROUP, ExhibitData.TeamEx
FROM (Teachers LEFT JOIN Persons AS Persons_1 ON Teachers.PID=Persons_1.PID) RIGHT JOIN ((Persons RIGHT JOIN Students ON Persons.PID=Students.PID) RIGHT JOIN (Schools RIGHT JOIN ((Chairs RIGHT JOIN JudgeTypes ON Chairs.CHAIRNUM=JudgeTypes.CHAIRNUM) RIGHT JOIN ((Categories RIGHT JOIN ExhibitData ON Categories.NCAT=ExhibitData.NCAT) LEFT JOIN ExhibitStudents ON ExhibitData.EXN=ExhibitStudents.EXN) ON JudgeTypes.JCAT=ExhibitData.JCAT) ON Schools.SCH_CODE=ExhibitData.SCH_CODE) ON Students.STID=ExhibitStudents.STID) ON Teachers.TID=ExhibitData.TID
WHERE (((ExhibitStudents.F_ORD)=1) AND ((ExhibitData.EXN)>100))
ORDER BY ExhibitData.EXN;
