SELECT ExhibitData.NCAT, Persons.LNAME, Persons.FNAME, Persons.MI, ExhibitData.JCAT, Judges.JUN, ExhibitJudges.EXN, ExhibitData.EXHB_TITLE, ExhibitData.FLOOR, ExhibitData.TeamEx, IIf(([UseTeamworkMark]=True And ([Team]=True Or [ExhibitData].[TeamEx]=-1)),"TEAM Exhibit: Consider Teamwork",IIf(([UseTeamworkMark]=False And ([Team]=True Or [ExhibitData].[TeamEx]=-1)),"TEAM Exhibit: Teamwork Mark Not Used","")) AS IsTeam, "<" & IIf(([UseTeamworkMark]=True And ([TEAM]=True Or [ExhibitData].[TeamEx]=-1)),"2","1") & Format([Exhibitdata].[EXN],"0000") & Format([Judges].[JUN],"0000") & [CKSUM] AS [BarCode-Old], IIf(([UseTeamworkMark]=True And ([TEAM]=True Or [ExhibitData].[TeamEx]=-1)),"2","1") & Format([Exhibitdata].[EXN],"0000") & Format([Judges].[JUN],"0000") AS [ID-Old], IIf([FLOOR]="Y","Floor Exhibit","") AS IsFloor, Categories.TEAM, Categories.CATNAME, Persons_1.LNAME AS ChairLName, Persons_1.FNAME AS ChairFName, "/" & Format(CInt(IIf(([UseTeamworkMark]=True And ([TEAM]=True Or [ExhibitData].[TeamEx]=-1)),"2","1"))+CInt(Mid(Format([Exhibitdata].[EXN],"0000"),1,1))+CInt(Mid(Format([Exhibitdata].[EXN],"0000"),2,1))+CInt(Mid(Format([Exhibitdata].[EXN],"0000"),3,1))+CInt(Mid(Format([Exhibitdata].[EXN],"0000"),4,1))+CInt(Mid(Format([Judges].[JUN],"0000"),1,1))+CInt(Mid(Format([Judges].[JUN],"0000"),2,1))+CInt(Mid(Format([Judges].[JUN],"0000"),3,1))+CInt(Mid(Format([Judges].[JUN],"0000"),4,1)),"00") & ">" AS [CKSum-Old], FairInfo.UseTeamworkMark, ExhibitJudges.FYR, IIf(([UseTeamworkMark]=True And ([Team]=True Or [ExhibitData].[TeamEx]=-1)),"Include Teamwork",IIf(([UseTeamworkMark]=False And ([Team]=True Or [ExhibitData].[TeamEx]=-1)),"","")) AS IsTeam2, IIf(([ExhibitData].[NCAT]="JENGR" Or [ExhibitData].[NCAT]="SENGR" Or [ExhibitData].[NCAT]="JTRAN" Or [ExhibitData].[NCAT]="STRAN" Or [ExhibitData].[NCAT]="JPHY" Or [ExhibitData].[NCAT]="SPHY" Or [ExhibitData].[NCAT]="JENGR" Or [ExhibitData].[NCAT]="JENGR"),"Engineering","Research") AS IsEngr, "2" & Format([Exhibitdata].[EXN],"0000") & Format([Judges].[JUN],"0000") AS ID, "<" & "2" & Format([Exhibitdata].[EXN],"0000") & Format([Judges].[JUN],"0000") & [CKSUM] AS BarCode, "/" & Format(CInt("2")+CInt(Mid(Format([Exhibitdata].[EXN],"0000"),1,1))+CInt(Mid(Format([Exhibitdata].[EXN],"0000"),2,1))+CInt(Mid(Format([Exhibitdata].[EXN],"0000"),3,1))+CInt(Mid(Format([Exhibitdata].[EXN],"0000"),4,1))+CInt(Mid(Format([Judges].[JUN],"0000"),1,1))+CInt(Mid(Format([Judges].[JUN],"0000"),2,1))+CInt(Mid(Format([Judges].[JUN],"0000"),3,1))+CInt(Mid(Format([Judges].[JUN],"0000"),4,1)),"00") & ">" AS CKSum, IIf([TeamEx]=False,[SForm].[Weight1] & "% " & [SForm].[Description1],IIf([Categories].[TmCritNum]<41,[SForm].[Weight1] & "% " & [Sform].[Description1],[SForm_1].[Weight1] & "% " & [Sform_1].[Description1])) AS Lbl1, IIf([TeamEx]=False,[SForm].[Weight2] & "% " & [SForm].[Description2],IIf([Categories].[TmCritNum]<41,[SForm].[Weight2] & "% " & [Sform].[Description2],[SForm_1].[Weight2] & "% " & [Sform_1].[Description2])) AS Lbl2, IIf([TeamEx]=False,[SForm].[Weight3] & "% " & [SForm].[Description3],IIf([Categories].[TmCritNum]<41,[SForm].[Weight3] & "% " & [Sform].[Description3],[SForm_1].[Weight3] & "% " & [Sform_1].[Description3])) AS Lbl3, IIf([TeamEx]=False,[SForm].[Weight4] & "% " & [SForm].[Description4],IIf([Categories].[TmCritNum]<41,[SForm].[Weight4] & "% " & [Sform].[Description4],[SForm_1].[Weight4] & "% " & [Sform_1].[Description4])) AS Lbl4, IIf([TeamEx]=False,[SForm].[Weight5] & "% " & [SForm].[Description5],IIf([Categories].[TmCritNum]<41,[SForm].[Weight5] & "% " & [Sform].[Description5],[SForm_1].[Weight5] & "% " & [Sform_1].[Description5])) AS Lbl5, IIf([TeamEx]=False,[SForm].[Weight6] & "% " & [SForm].[Description6],IIf([Categories].[TmCritNum]<41,[SForm].[Weight6] & "% " & [Sform].[Description6],[SForm_1].[Weight6] & "% " & [Sform_1].[Description6])) AS Lbl6
FROM (((Categories LEFT JOIN SForm ON Categories.CritNum=SForm.CritNum) LEFT JOIN SForm AS SForm_1 ON Categories.TmCritNum=SForm_1.CritNum) RIGHT JOIN ExhibitData ON Categories.NCAT=ExhibitData.NCAT) RIGHT JOIN (((Chairs LEFT JOIN Judges AS Judges_1 ON Chairs.JUN=Judges_1.JUN) LEFT JOIN Persons AS Persons_1 ON Judges_1.PID=Persons_1.PID) RIGHT JOIN ((Persons RIGHT JOIN (JudgeTypes RIGHT JOIN (Judges RIGHT JOIN ExhibitJudges ON Judges.JUN=ExhibitJudges.JUN) ON JudgeTypes.JCAT=Judges.JCAT) ON Persons.PID=Judges.PID) LEFT JOIN FairInfo ON ExhibitJudges.FYR=FairInfo.FAIR_YEAR) ON Chairs.CHAIRNUM=JudgeTypes.CHAIRNUM) ON ExhibitData.EXN=ExhibitJudges.EXN
ORDER BY ExhibitData.NCAT, Persons.LNAME, Persons.FNAME, ExhibitData.JCAT, ExhibitJudges.EXN;
